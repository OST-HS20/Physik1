\begin{formula}
	{F = \frac{p}{t} \rightarrow p = m \cdot v} 
	
	F & Kraft & [N] \\
	p & Impuls & [Ns, kg \cdot m/s] \\
	t & Dauer der Kraft & [s] \\
	m & Masse & [kg] \\
	v & Geschw. & [m/s]
\end{formula}

\begin{formula}

	v & Geschw. \underline{vor} Stoss & [m/s] \\
	v' & Geschw. \underline{nach} Stoss & [m/s] \\
\end{formula}
